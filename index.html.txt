  <style>
    body {
      display: flex;
      justify-content: center;
      align-items: center;
      height: 100vh;
      margin: 0;
      background-color: #f0f0f0;

    }
    #game-container {
      text-align: center;
    }
    canvas {
      border: 2px solid #333;
    }
    #start-screen, #game-over-screen {
      position: absolute;
      top: 50%;
      left: 50%;
      transform: translate(-50%, -50%);
      background: rgba(255, 255, 255, 0.9);
      padding: 20px;
      border-radius: 10px;
      box-shadow: 0 0 10px rgba(0, 0, 0, 0.3);
    }
    select, button {
      margin: 10px;

      font-size: 16px;
    }
    button {
      cursor: pointer;
      background-color: #4CAF50;

      background-color: #45a049;

      color: #333;

<body>
  <div id="game-container"></div>
  <script>
    let gameState = 'start';
    let speed, difficulty, fallSpeed;
    let words = [];
    let score = 0;
    let level = 1;
    let eliminatedWords = 0;
    let inputBuffer = '';
    
    // 倉頡碼資料庫（簡化版，實際應用需更大資料庫）
    const cangjieDB = [
      { char: '日', code: 'A', level: 1 },
      { char: '月', code: 'B', level: 1 },
      { char: '口', code: 'R', level: 1 },
      { char: '田', code: 'W', level: 2 },
      { char: '好', code: 'VND', level: 5 },
      { char: '明', code: 'AB', level: 3 },
      { char: '星', code: 'AP', level: 4 },
      { char: '機', code: 'MGRR', level: 10 },
      { char: '電', code: 'MFB', level: 8 },
      { char: '腦', code: 'YTBU', level: 10 }
    ];
    
    function setup() {
      let canvas = createCanvas(600, 800);
      canvas.parent('game-container');
      textAlign(CENTER, CENTER);
      textSize(32);
    }
    
    function draw() {
      background(220);
      
      if (gameState === 'start') {
        drawStartScreen();
      } else if (gameState === 'playing') {
        drawGame();
      } else if (gameState === 'gameOver') {
        drawGameOverScreen();
      }
    }
    
    function drawStartScreen() {
      let startDiv = select('#start-screen');
      if (!startDiv) {
        startDiv = createDiv().id('start-screen');
        startDiv.html(`
          <h1>倉頡方塊</h1>
          <p>選擇遊戲速度和難度，輸入倉頡碼消除跌落文字！</p>
          <label>速度：</label>
          <select id="speed-select">
            <option value="1">慢速</option>
            <option value="2">中速</option>
            <option value="3">快速</option>
          </select><br>
          <label>難度：</label>
          <select id="difficulty-select">
            <option value="1">簡單 (LV.1)</option>
            <option value="5">中等 (LV.5)</option>
            <option value="10">困難 (LV.10)</option>
          </select><br>
          <button onclick="startGame()">開始</button>
        `);
      }
    }
    
    function startGame() {
      speed = parseInt(select('#speed-select').value());
      difficulty = parseInt(select('#difficulty-select').value());
      fallSpeed = speed * 2; // 每秒下落格數
      words = [];
      score = 0;
      level = 1;
      eliminatedWords = 0;
      inputBuffer = '';
      select('#start-screen').remove();
      gameState = 'playing';
      spawnWord();
    }
    
    function drawGame() {
      // 繪製遊戲區域
      fill(255);
      rect(0, 0, width, height);
      
      // 更新和繪製文字
      for (let i = words.length - 1; i >= 0; i--) {
        let word = words[i];
        if (!word.isStatic) {
          word.y += fallSpeed / frameRate();
        }
        
        // 檢查是否到底部
        if (word.y >= height - 40 && !word.isStatic) {
          word.isStatic = true;
          spawnWord();
        }
        
        // 檢查遊戲結束
        if (word.y <= 40 && word.isStatic) {
          gameState = 'gameOver';
        }
        
        // 繪製文字
        fill(word.isStatic ? 100 : 0);
        text(word.char, word.x, word.y);
      }
      
      // 顯示資訊
      fill(0);
      textSize(20);
      text(`分數: ${score} 關卡: ${level} 輸入: ${inputBuffer}`, width / 2, 30);
      
      // 每隔3秒生成新文字
      if (frameCount % (180 / speed) === 0) {
        spawnWord();
      }
    }
    
    function spawnWord() {
      let maxLevel = difficulty === 1 ? 3 : (difficulty === 5 ? 7 : 10);
      let availableWords = cangjieDB.filter(w => w.level <= maxLevel && w.level >= level - 2 && w.level <= level);
      if (availableWords.length === 0) {
        availableWords = cangjieDB.filter(w => w.level <= maxLevel);
      }
      let word = random(availableWords);
      words.push({
        char: word.char,
        code: word.code,
        x: random(50, width - 50),
        y: 40,
        isStatic: false
      });
    }
    
    function drawGameOverScreen() {
      let gameOverDiv = select('#game-over-screen');
      if (!gameOverDiv) {
        gameOverDiv = createDiv().id('game-over-screen');
        gameOverDiv.html(`
          <h1>遊戲結束</h1>
          <p>最終分數: ${score}</p>
          <p>最終關卡: ${level}</p>
          <button onclick="restartGame()">重新開始</button>
        `);
      }
    }
    
    function restartGame() {
      select('#game-over-screen').remove();
      gameState = 'start';
    }
    
    function keyPressed() {
      if (gameState === 'playing') {
        if (keyCode === BACKSPACE) {
          inputBuffer = inputBuffer.slice(0, -1);
        } else if (keyCode === ENTER) {
          checkInput();
        }
      }
    }
    
    function keyTyped() {
      if (gameState === 'playing' && key.match(/[a-z]/i)) {
        inputBuffer += key.toUpperCase();
      }
    }
    
    function checkInput() {
      let lowestWord = null;
      let minY = -Infinity;
      for (let word of words) {
        if (!word.isStatic && word.y > minY) {
          lowestWord = word;
          minY = word.y;
        }
      }
      
      if (lowestWord && inputBuffer === lowestWord.code) {
        words.splice(words.indexOf(lowestWord), 1);
        score += 100;
        eliminatedWords++;
        inputBuffer = '';
        
        // 檢查關卡升級
        if (eliminatedWords >= 10) {
          level = min(level + 1, 10);
          eliminatedWords = 0;
        }
      } else {
        inputBuffer = ''; // 錯誤輸入清空
      }
    }
  </script>
